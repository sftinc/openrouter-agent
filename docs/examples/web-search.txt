/**
 * search_web — Search the public web for information not found in prior tickets.
 *
 * Only invoked when the user explicitly asks the agent to search the web or look
 * something up online. Each query is run through a parallel sub-agent call that
 * uses OpenRouter's `openrouter:web_search` server tool (engine: exa) to execute
 * the search server-side, then the sub-agent returns a structured JSON payload
 * with `{ answer, citations }`.
 *
 * Mirrors the sub-agent shape used by `search-prior-tickets.ts` — both tools delegate
 * structured extraction to `deps.complete()` rather than hitting OpenRouter directly.
 *
 * Input (from the LLM tool call):
 *   { "queries": ["how to cancel an eSnipe snipe after auction end"] }
 *
 * Output (returned to the LLM as a JSON string):
 *   {
 *     "results": [
 *       {
 *         "query": "how to cancel an eSnipe snipe after auction end",
 *         "answer": "Once an auction has ended, snipes cannot be cancelled...",
 *         "citations": [
 *           { "title": "eSnipe Help — Cancelling Snipes", "url": "https://..." }
 *         ]
 *       }
 *     ]
 *   }
 */
import type { ToolDef, AgentDeps } from '../types'

// --- Tool definition ---

export const searchWebToolDef: ToolDef = {
	type: 'function',
	function: {
		name: 'search_web',
		description:
			'Search the public web and return a synthesized answer plus citations per query. Only call when the user explicitly asks to search the web or look something up online — never as a fallback for search_prior_tickets.',
		parameters: {
			type: 'object',
			properties: {
				queries: {
					type: 'array',
					items: { type: 'string' },
					description: '1-3 focused web search queries. Usually one is enough.',
					minItems: 1,
					maxItems: 3,
				},
			},
			required: ['queries'],
		},
	},
}

// --- Types ---

export interface SearchWebResult {
	results: Array<{
		query: string
		answer: string
		citations: Array<{ title: string; url: string }>
	}>
}

export interface SearchWebDeps {
	complete: AgentDeps['complete']
}

interface SubAgentResponse {
	answer: string
	citations: Array<{ title: string; url: string }>
}

// --- Sub-agent schema ---

const subAgentSchema = {
	type: 'object',
	properties: {
		answer: {
			type: 'string',
			description:
				'Detailed prose answer with inline markdown links tying each claim to its source. Prefer verbatim steps, labels, and quoted text over paraphrase — detail beats brevity.',
		},
		citations: {
			type: 'array',
			description: 'Sources used to construct the answer.',
			items: {
				type: 'object',
				properties: {
					title: { type: 'string', description: 'Page title of the cited source.' },
					url: { type: 'string', description: 'URL of the cited source.' },
				},
				required: ['title', 'url'],
				additionalProperties: false,
			},
		},
	},
	required: ['answer', 'citations'],
	additionalProperties: false,
} as const

const SUB_AGENT_SYSTEM_PROMPT = `You are a web research assistant serving a support-agent drafting tool. Use the openrouter:web_search tool to gather information on the query, then return a JSON object with \`answer\` and \`citations\`.

Your answer goes downstream to a drafting agent that will distill it for a customer reply, so detail beats brevity. Capture specific steps, exact UI labels, error text, version numbers, and direct quotes where they clarify meaning. Use inline markdown links to tie each claim to its source (e.g. "click Settings → Billing per [the docs](https://…)").

Rules:
- Ground every factual statement in content the web_search tool returned. Do not draw on prior knowledge. Never invent a citation.
- If sources disagree, say so and show both positions.
- If the search returns nothing useful, set \`answer\` to a short note that no good sources were found and return an empty \`citations\` array.

The \`citations\` array is the deduplicated list of sources you used, each with \`title\` and \`url\`.`

// --- Execute ---

export async function executeSearchWeb(args: { queries: string[] }, deps: SearchWebDeps): Promise<SearchWebResult> {
	// Each query runs independently. If one fails (API error, malformed JSON,
	// etc.) we degrade to a failure-marked result instead of rejecting the
	// whole Promise.all — that way the agent sees which queries succeeded and
	// which didn't, and can respond to the user gracefully (retry, fall back
	// to prior-ticket context, or explain the failure) instead of the whole
	// turn erroring out.
	const results = await Promise.all(
		args.queries.map(async (query) => {
			try {
				const response = await deps.complete(
					[
						{ role: 'system', content: SUB_AGENT_SYSTEM_PROMPT },
						{ role: 'user', content: query },
					],
					{
						temperature: 0.2,
						model: 'anthropic/claude-haiku-4.5',
						tools: [{ type: 'openrouter:web_search', engine: 'exa', max_results: 5 }],
						responseFormat: {
							type: 'json_schema',
							schema: subAgentSchema as unknown as Record<string, unknown>,
						},
						purpose: 'search_web',
					},
				)

				const raw = response.content ?? '{"answer":"","citations":[]}'
				const parsed = JSON.parse(raw) as SubAgentResponse
				return { query, answer: parsed.answer, citations: parsed.citations }
			} catch (err) {
				const reason = (err as Error).message ?? String(err)
				console.error('[search-web] query failed', { query, reason })
				return {
					query,
					answer: `Web search failed for this query: ${reason}. Let the user know and offer to try again or use another source.`,
					citations: [],
				}
			}
		}),
	)
	return { results }
}
